#include <iostream>
#include <clocale>
using namespace std;
void menu()
{
	cout << "Какое действие хочешь выполнить?" << endl;
	cout << "+" << endl;
	cout << "-" << endl;
	cout << "/" << endl;
	cout << "*" << endl;
}
void calc(int x,int y)
{
	char del;
	float z;
	menu();
	cin >> del;
	switch (del)
	{
	case'+':
			z = x + y;
			break;
	case'*':
		z = x * y;
		break;
	case'/':
		z = x / y;
		break;
	case'-':
		z = x - y;
		break;
	}
	cout << "Result:" << z << endl;
}
